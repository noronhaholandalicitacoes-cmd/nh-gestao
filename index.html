'dd/MM/yyyy HH:mm' : 'dd/MM/yyyy'; clean.push(Utilities.formatDate(c, Session.getScriptTimeZone(), fmt)); } else { clean.push(c === null || c === undefined ? '' : c); } } // Para EM ANDAMENTO sem ID: usar índice da linha como identificador virtual // O sistema vai usar rowKey = empresa+licitacao para encontrar a linha result.push(clean); } out.setContent(JSON.stringify({ values: result, total: result.length })); return out; } // ── APPEND ──────────────────────────────────────────────────── function doAppend(out, sheet, sheetName, p) { if (sheetName === 'FINALIZADOS') { out.setContent(JSON.stringify({ error: 'FINALIZADOS gerenciado por formula.' })); if (n > maxId) maxId = n; } nextId = maxId + 1; } if (sheetName === 'FINANCEIRO') { // Financeiro: dados começam na linha 7, inserir no final var finLastRow = sheet.getLastRow(); var insertRow = finLastRow + 1;
